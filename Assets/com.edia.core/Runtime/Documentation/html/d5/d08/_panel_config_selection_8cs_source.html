140 private bool ValidateConfigs() {
141 // Validate all strings on formatting
142
143 if (!ValidateJSON(_sessionInfoJsonString)) {
144 Debug.Log("JSON Format error: Session Info");
145 EventManager.TriggerEvent(Edia.Events.ControlPanel.EvShowMessageBox, new eParam("JSON Format error: session_info.json", false ));
146 return false;
147 }
148
149 if (!ValidateJSON(_xBlockSequenceJsonString)) {
150 Debug.Log("JSON Format error: Xblock sequence");
151 EventManager.TriggerEvent(Edia.Events.ControlPanel.EvShowMessageBox, new eParam("JSON Format error: session_sequence.json", false));
152 return false;
153 }
154
155 foreach (string s in _taskDefinitionJsonStrings) {
156 if (!ValidateJSON(s)) {
157 Debug.Log("JSON Format error: Task definition");
158 EventManager.TriggerEvent(Edia.Events.ControlPanel.EvShowMessageBox, new eParam("JSON Format error: Task definition", false));
159 return false;
160 }
161 }
162
163 foreach (string s in _xBlockDefinitionJsonStrings) {
164 if (!ValidateJSON(s)) {
165 Debug.Log("JSON Format error: Xblock definition");
166 EventManager.TriggerEvent(Edia.Events.ControlPanel.EvShowMessageBox, new eParam("JSON Format error: Xblock definition", false));
167 return false;
168 }
169 }
170
171 // Sanity check - for each entry in the sequence file, there should be a file with the same name
172 _xBlockSequence = UnityEngine.JsonUtility.FromJson<XBlockSequence>(_xBlockSequenceJsonString);
173
174 foreach (string s in _xBlockSequence.Sequence) {
175 bool found = false;
176 //Debug.Log(" src: " + s.ToLower() );
177 foreach ( string deffile in xBlockDefinitionsFileList ) {
178 //Debug.Log(" check: " + deffile.Split('.')[0].ToLower());
179 if ( s.ToLower() == deffile.Split('.')[0].ToLower() ) {
180 found = true;
181 break;
182 }
183 }
184 if (!found) {
185 Debug.LogError("No <b>" + s + ".json</b> config file not found in xblock definitions folder");
186 return false;
187 }
188 }
189
190 return true;
191 }
