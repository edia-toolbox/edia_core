349 void BlockStart() {
350 AddToLog("Block Start");
351
352 // Set new storedBlockNum value
353 _activeSessionBlockNum = Session.instance.currentBlockNum;
354 _activeXBlock = XBlocks[XBlocks.FindIndex(x => x.name == Session.instance.CurrentBlock.settings.GetString("_assetId"))];
355 _activeXBlock.enabled = true;
356 _activeXBlock.gameObject.SetActive(true);
357 _activeXBlock.OnBlockStart();
358
359 // Update block progress
360 UpdateBlockProgress();
361
362 // Check for block introduction flag
363 bool hasIntro = Session.instance.CurrentBlock.settings.GetStringList("_start").Count > 0;
364
365 // Inject introduction step or continue UXF sequence
366 if (hasIntro) {
367 EventManager.StartListening(Edia.Events.StateMachine.EvProceed, BlockContinueAfterIntro); // listener as it event call can come from any script
368 ShowMessageToUser (Session.instance.CurrentBlock.settings.GetStringList("_start"));
369 UpdateProgressStatus(Session.instance.CurrentBlock.settings.GetString("blockId") + " Introduction");
370 }
371 else {
372 StartTrial();
373 UpdateProgressStatus(Session.instance.CurrentBlock.settings.GetString("blockId"));
374 }
375 }
376
377 void BlockEnd() {
378 _activeXBlock.OnBlockEnd();
379
380 // Check for block outro flag
381 bool hasOutro = Session.instance.CurrentBlock.settings.GetStringList("_end").Count > 0;
382
383 if (hasOutro) {
384 EventManager.StartListening(Edia.Events.StateMachine.EvProceed, BlockContinueAfterOutro); // listener as it event call can come from any script
385 EnableProceedButton(true);
386 ShowMessageToUser(Session.instance.CurrentBlock.settings.GetStringList("_end"));
387 UpdateProgressStatus("Block Outro");
388 _activeXBlock.OnBlockOutro();
389 }
390 else {
391 BlockCheckAndContinue();
392 }
393 }
394
395 void BlockCheckAndContinue() {
396
397 _activeXBlock.enabled = false;
398 _activeXBlock.gameObject.SetActive(false);
399
400 // Is this then the last trial of the session?
401 if (Session.instance.LastTrial == Session.instance.CurrentTrial) {
402 AddToLog("Reached end of trials ");
403 FinalizeSession();
404 return;
405 }
406
407 Session.instance.BeginNextTrialSafe();
408 }

442 void OnTrialBeginUXF(Trial newTrial) {
443 AddToExecutionOrderLog("OnTrialBeginUXF");
444
445 XRManager.Instance.ShowVRInstantly(); // TODO: This is now assuming that we want VR visable at the start of each trial.
446
447 bool isNewBlock = (Session.instance.currentBlockNum != _activeSessionBlockNum) && (Session.instance.currentBlockNum <= Session.instance.blocks.Count);
448
449 if (isNewBlock) {
450 BlockStart();
451 }
452 else {
453 StartTrial();
454 UpdateProgressStatus(Session.instance.CurrentBlock.settings.GetString("blockId"));
455 }
456 }
457
459 void OnTrialEndUXF(Trial endedTrial) {
460 AddToExecutionOrderLog("OnTrialEnd");
461 SaveCustomDataTables();
462
463 _activeXBlock.OnEndTrial();
464
465 // Are we ending?
466 if (Session.instance.isEnding)
467 return;
468
469 // Is there a PAUSE requested right now?
470 if (_isPauseRequested) {
471 _isPauseRequested = false;
472
473 if (endedTrial == Session.instance.LastTrial)
474 return;
475
476 AddToExecutionOrderLog("Injected SessionBreak");
477 SessionPause();
478 return;
479 }
480
481 // Reached last trial in a block?
482 if (Session.instance.CurrentBlock.lastTrial != endedTrial) { // NO
483 Session.instance.BeginNextTrialSafe();
484 return;
485 }
486 else {
487 BlockEnd(); // YES
488 return;
489 }
490 }

530 void NextTrialStep() {
531 if (ShowLog) AddToLog("Nextstep >");
532
533 if (_proceedTimer != null) {
534 StopCoroutine(_proceedTimer); // Kill timer, if any
535 EventManager.TriggerEvent(Edia.Events.ControlPanel.EvStopTimer);

540

544 InBetweenSteps();
545 UpdateStepProgress();
546 _activeXBlock.trialSteps[_currentStep].Invoke();
547
548 }
549 else EndTrial();
550 }

595 public void ConvertAndSaveDataToUXF(string[] headers, List<string[]> values, string filename) {
596 var UXFheaders = headers;
597 var data = new UXF.UXFDataTable(UXFheaders);
598
599 foreach (string[] valuerow in values) {
600 UXFDataRow newRow = new UXFDataRow();
601 for (int s = 0; s < valuerow.Length; s++) {
602 newRow.Add((UXFheaders[s], valuerow[s]));
603 }
604 data.AddCompleteRow(newRow);
605 }
606
607 // Save data
608 Session.instance.CurrentTrial.SaveDataTable(data, filename);
609 }
