142 public void GenerateUxfSequence() {
143 // First validate
144 if (!ValidateBlockList()) {
145 Debug.LogError("Supplied blocklist is invalid");
146 return;
147 }
148
149 // Loop through BlockList, create blocks
150 foreach (var blockId in _xBlockSequence.Sequence) {
151
152 Block newBlock = Session.instance.CreateBlock();
153
154 // Find the according XBlockBase (e.g., Task or Break definition) and get global settings
155 XBlockBaseSettings xBlockBase = GetXBlockBaseByBlockId(blockId);
156 if (xBlockBase == null) {
157 Debug.LogError($"Failed getting details for {blockId} ");
158 return;
159 }
160
161 if (xBlockBase.settings.Count > 0) {
162 newBlock.settings.UpdateWithDict(Helpers.GetSettingsTupleListAsDict(xBlockBase.settings));
163 }
164
165 newBlock.settings.SetValue("_start", GetValuesListByKey(xBlockBase.instructions, "_start")); //
166 newBlock.settings.SetValue("_end", GetValuesListByKey(xBlockBase.instructions, "_end")); //
167
168 XBlockSettings currentXBlock = GetXBlockByBlockId(blockId);
169 newBlock.settings.SetValue("blockType", currentXBlock.type.ToLower());
170 newBlock.settings.SetValue("blockId", currentXBlock.blockId.ToLower());
171 newBlock.settings.SetValue("_assetId", currentXBlock.type.ToLower() + "_" + currentXBlock.subType.ToLower());
172
173 // Add block specific instructions, if any
174 foreach (string s in new string[] { "_start", "_end" }) {
175 List<string> newList = newBlock.settings.GetStringList(s);
176 newList.AddRange(GetValuesListByKey(currentXBlock.instructions, s));
177 newBlock.settings.SetValue(s, newList);
178 }
179
180 // Continue with settings
181 newBlock.settings.UpdateWithDict(Helpers.GetSettingsTupleListAsDict(currentXBlock.settings)); // add block specific settings
182
183 // Add settings and trials specific for Break vs Task XBlocks
184 string currentXBlockType = GetXBlockType(blockId);
185
186 switch (currentXBlockType.ToLower()) {
187 case "break":
188 newBlock.settings.SetValue("subType", currentXBlock.subType);
189 newBlock.settings.SetValue("_info", GetValuesListByKey(currentXBlock.instructions, "_info"));
190 newBlock.CreateTrial(); // create 1 dummy trial
191 break;
192
193 case "task":
194 newBlock.settings.SetValue("subType", currentXBlock.subType);
195
196 // Add trials
197 foreach (ValueList row in currentXBlock.trialSettings.valueList) {
198 Trial trial = newBlock.CreateTrial();
199
200 for (int i = 0; i < row.values.Count; i++) {
201 trial.settings.SetValue(currentXBlock.trialSettings.keys[i], row.values[i]); // set values to trial
202 }
203 }
204
205 // Log all unique TRIAL settings keys
206 foreach (string k in currentXBlock.trialSettings.keys) {
207 if (IsValidKeyForTrialResults(k))
208 Session.instance.settingsToLog.Add(k);
209 }
210 break;
211
212 default:
213 Debug.LogError($"XBlock type must be either 'Task' or 'Break'; cannot be '{currentXBlockType}'.");
214 break;
215 }
216
217 // Log all unique and valid TASK setting keys
218 foreach (string k in newBlock.settings.Keys) {
219 if (IsValidKeyForTrialResults(k))
220 Session.instance.settingsToLog.Add(k);
221 }
222 }
223 }
224 }
