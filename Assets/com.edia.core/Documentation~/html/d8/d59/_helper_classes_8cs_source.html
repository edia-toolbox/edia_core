92 public static Dictionary<string, object> GetSettingsTupleListAsDict(List<SettingsTuple> list) {
93 Dictionary<string, object> tmp = new Dictionary<string, object>();
94 foreach (SettingsTuple st in list)
95 if (st.value.Contains(',')) { // it's a list!
96 List<string> stringlist = st.value.Split(',').ToList();
97 for (int s = 0; s < stringlist.Count; s++) {
98 string newstring = stringlist[s].Replace(" ", string.Empty); // remove spaces
99 stringlist[s] = newstring;
100 }
101 tmp.Add(st.key, stringlist);
102 }
103 else tmp.Add(st.key, st.value); // normal string
104
105 return tmp;
106 }
107 }
